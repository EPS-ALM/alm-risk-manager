Anlise de Risco de Investimentos

1. Introduo

Neste notebook, realizaremos uma anlise de risco para investimentos em ativos financeiros do mercado brasileiro. A anlise utiliza:

  • Carteira de Pesos Iguais: Onde todos os ativos possuem o mesmo peso no portflio (para comparao).
  • Carteira Otimizada (Markowitz): Otimizada para minimizar a volatilidade com base na matriz de covarincia dos retornos.
  • Comparao com o ndice IBOVESPA (BOVA11): Usado como benchmark.

Metodologia

Os clculos realizados incluem:

  1. Retornos Dirios: Calculados com base nos preos ajustados dos ativos.
  2. Volatilidade Anualizada: Usada para avaliar o risco.
  3. Carteira Otimizada: Calculada utilizando o mtodo de Markowitz para encontrar a combinao ideal de ativos que minimize o risco.
  4. Retorno Acumulado: Mostra a performance dos portflios no perodo selecionado.

O notebook tambm permite ao usurio escolher:

  • Os ativos para compor o portflio.
  • O perodo de anlise: 1M (1 ms), 6M (6 meses), 1Y (1 ano), 3Y (3 anos) ou 5Y (5 anos).

Objetivo

Este notebook ajuda investidores a visualizar e comparar a performance de diferentes estratgias de alocao de ativos. Os resultados so apresentados graficamente para facilitar a anlise.

def definir_periodo(escolha):
    """Define o perodo de anlise com base na escolha do usurio."""
    fim = datetime.now()
    
    # Processar entrada em meses (M) ou anos (Y)
    if escolha.endswith("M"):  # Entrada em meses
        meses = int(escolha[:-1])
        if meses > 60:  # Limitar a 60 meses (5 anos)
            raise ValueError("O perodo mximo permitido  de 60 meses (5 anos).")
        inicio = fim - timedelta(days=30 * meses)
    
    elif escolha.endswith("Y"):  # Entrada em anos
        anos = int(escolha[:-1])
        if anos > 5:  # Limitar a 5 anos
            raise ValueError("O perodo mximo permitido  de 5 anos.")
        inicio = fim - timedelta(days=365 * anos)
    
    else:
        raise ValueError("Escolha invlida! Use o formato '<n>M' para meses ou '<n>Y' para anos.")
    
    return inicio, fim

def baixar_dados(escolhas, periodo):
    inicio, fim = definir_periodo(periodo)
    ativos_selecionados = [ativos[ativo] for ativo in escolhas]
    dados = yf.download(ativos_selecionados, start=inicio, end=fim.strftime('%Y-%m-%d'))["Adj Close"]
    dados.columns = escolhas
    return dados.dropna()

def calcular_pesos_iguais(retornos):
    num_ativos = len(retornos.columns)
    pesos = np.array([1 / num_ativos] * num_ativos)
    retorno_carteira = (retornos @ pesos).cumsum()
    return pesos, retorno_carteira
  • Carteira Otimizada (Markowitz)
In[56]:
def calcular_carteira_otimizada(retornos):
    cov_matrix = retornos.cov() * 252
    num_ativos = len(retornos.columns)

    def objetivo(pesos):
        return np.sqrt(np.dot(pesos.T, np.dot(cov_matrix, pesos)))

    restricao = {'type': 'eq', 'fun': lambda x: np.sum(x) - 1}
    limites = tuple((0, 1) for _ in range(num_ativos))
    pesos_iniciais = np.array([1 / num_ativos] * num_ativos)

    resultado = minimize(objetivo, pesos_iniciais, method="SLSQP", bounds=limites, constraints=restricao)
    pesos = resultado.x
    retorno_carteira = (retornos @ pesos).cumsum()
    return pesos, retorno_carteira

def gerar_matriz_correlacao(retornos):
    """Gera a matriz de correlao e a exibe como um heatmap."""
    correlacao = retornos.corr()
    plt.figure(figsize=(10, 8))
    sns.heatmap(correlacao, annot=True, cmap="coolwarm", fmt=".2f", cbar=True)
    plt.title("Matriz de Correlao dos Ativos Selecionados")
    plt.show()
    return correlacao

def gerar_grafico(retorno_iguais, retorno_otimizado, retorno_ibov, periodo_escolhido):
    """Gera o grfico de comparao de retornos acumulados com valores finais na legenda."""
    plt.figure(figsize=(14, 8))
    
    # Valores finais
    final_values = {
        "Carteira Otimizada": (retorno_otimizado.iloc[-1].item(), "green"),
        "Carteira Pesos Iguais": (retorno_iguais.iloc[-1].item(), "lightgreen"),
        "IBOVESPA (BOVA11)": (retorno_ibov.iloc[-1].item(), "blue"),
    }

    # Plotar os retornos acumulados
    plt.plot(retorno_otimizado, label=f"Carteira Otimizada ({final_values['Carteira Otimizada'][0]:+.2%})", color="green")
    plt.plot(retorno_iguais, label=f"Carteira Pesos Iguais ({final_values['Carteira Pesos Iguais'][0]:+.2%})", color="lightgreen", linestyle="--")
    plt.plot(retorno_ibov, label=f"IBOVESPA (BOVA11) ({final_values['IBOVESPA (BOVA11)'][0]:+.2%})", color="blue")
    
    # Linha no eixo zero
    plt.axhline(0, color="gray", linestyle="--", linewidth=1, alpha=0.8)
    
    # Configurao do eixo Y (em porcentagem)
    plt.gca().yaxis.set_major_formatter(plt.FuncFormatter(lambda y, _: f"{y:.0%}"))
    
    # Ajuste dinmico do eixo X com base no perodo
    if periodo_escolhido.endswith("M"):
        meses = int(periodo_escolhido[:-1])
        if meses <= 12:  # At 12 meses
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=1))  # Marcao mensal
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        elif meses <= 24:  # Entre 12 meses e 2 anos
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=3))  # A cada 3 meses
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        elif meses <= 36:  # Entre 2 anos e 3 anos
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=6))  # A cada 6 meses
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        else:  # Mais de 3 anos
            plt.gca().xaxis.set_major_locator(mdates.YearLocator())  # Marcao anual
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y'))
    
    elif periodo_escolhido.endswith("Y"):
        anos = int(periodo_escolhido[:-1])
        if anos == 1:  # At 1 ano
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=1))  # Marcao mensal
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        elif anos <= 2:  # At 2 anos
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=3))  # A cada 3 meses
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        elif anos <= 3:  # At 3 anos
            plt.gca().xaxis.set_major_locator(mdates.MonthLocator(interval=6))  # A cada 6 meses
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
        else:  # Mais de 3 anos
            plt.gca().xaxis.set_major_locator(mdates.YearLocator())  # Marcao anual
            plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y'))

    # Adicionar rtulos finais no grfico
    for label, (value, color) in final_values.items():
        bbox_props = dict(
            boxstyle="round,pad=0.3", 
            edgecolor="none", 
            facecolor=color, 
            alpha=0.6
        )
        plt.text(retorno_otimizado.index[-1], value, f"{value:+.2%}",
                 fontsize=12, color="white", ha="left", va="center", bbox=bbox_props)

    # Configurao final do grfico
    plt.title("Comparao de Carteiras")
    plt.xlabel("Perodo")
    plt.ylabel("Retorno Acumulado (%)")
    plt.legend(loc="upper left")  # Legenda movida para o canto superior esquerdo
    plt.grid()
    plt.show()
